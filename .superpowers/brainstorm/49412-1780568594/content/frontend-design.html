ServisEkrani — Matching Mantığı

// Haversine (iki nokta arası km)

// Adım 1: Kategoriye göre filtrele
const eslesmis = servisler.filter(s =>
  s.kategoriler.includes(cihaz)
);

// Adım 2: Kullanıcı konumu al
navigator.geolocation.getCurrentPosition(pos => {
  const { latitude: lat, longitude: lng } = pos.coords;

  // Adım 3: Mesafe hesapla + sırala
  const sirali = eslesmis
    .map(s => ({ ...s, km: mesafe(lat, lng, s.lat, s.lng) }))
    .sort((a, b) => a.km - b.km)
    .slice(0, 10);   // ilk 10
  setSiraliServisler(sirali);
});
